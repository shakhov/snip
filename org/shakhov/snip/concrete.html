1.1.1 –û—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–∞—è –≤—ã—Å–æ—Ç–∞ —Å–∂–∞—Ç–æ–π –∑–æ–Ω—ã

–û—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–∞—è –≤—ã—Å–æ—Ç–∞ —Å–∂–∞—Ç–æ–π –∑–æ–Ω—ã \(\xi = x/h_0\) –Ω–µ –¥–æ–ª–∂–Ω–∞ –ø—Ä–µ–≤—ã—à–∞—Ç—å –ø—Ä–µ–¥–µ–ª—å–Ω–æ–π –≤–µ–ª–∏—á–∏–Ω—ã \(\xi_y\)

\begin{equation} \xi_y = \cfrac{\omega}{1 + \frac{\sigma_1}{\sigma_2}\left(1-\frac{\omega}{1.1}\right)} \end{equation} 
:xi<=xi-y
(fnk
 ""
 [xi xi-y]
 (<= xi xi-y))

:xi
(fnk
 ""
 [h0 x]
 (/ x h0))

:xi-y
(fnk
 ""
 [omega sigma-1 sigma-2]
 (/ omega
    (+ 1 (* (/ sigma-1 sigma-2)
            (- 1 (/ omega 1.1))))))

1.2 –†–∞—Å—á—ë—Ç –ø—Ä—è–º–æ—É–≥–æ–ª—å–Ω—ã—Ö —Å–µ—á–µ–Ω–∏–π

1.2.1 \(M_{max}\) — –ø—Ä–µ–¥–µ–ª—å–Ω—ã–π –º–æ–º–µ–Ω—Ç

–ü—Ä–µ–¥–µ–ª—å–Ω–æ –¥–æ–ø—É—Å—Ç–∏–º—ã–π –º–æ–º–µ–Ω—Ç –ø—Ä–∏ –∏–∑–≥–∏–±–µ –ø—Ä—è–º–æ—É–≥–æ–ª—å–Ω—ã—Ö –∂–µ–ª–µ–∑–æ–±–µ—Ç–æ–Ω–Ω—ã—Ö —Å–µ—á–µ–Ω–∏–π –≤—ã—á–∏—Å–ª—è–µ—Ç—Å—è –ø–æ —Ñ–æ—Ä–º—É–ª–µ:

\begin{equation} M \le M_{max} = R_b b x (h_0 - 0.5 x) + R_{sc} A_{sc,ef} (h_{01} - a_{sc}) + \sigma_{pc} A_pc (h_0 - a_p) \end{equation} 
:M-max
(fnk
 ""
 [Rb b x h0 h01 Asc-ef Rsc asc sigma-pc Apc apc]
 (+ (* Rb b x
       (- h0 (* 0.5 x)))
    (* Rsc Asc-ef
       (- h01 asc))
    (* sigma-pc Apc
       (- h0 apc))))

1.2.2 \(x\) — –≤—ã—Å–æ—Ç–∞ —Å–∂–∞—Ç–æ–π –∑–æ–Ω—ã

–í—ã—Å–æ—Ç–∞ —Å–∂–∞—Ç–æ–π –∑–æ–Ω—ã –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç—Å—è –ø–æ —Ñ–æ—Ä–º—É–ª–µ:

\begin{equation} x = \frac{R_p A_p + R_s A_s - R_{sc} A_{sc,ef} - \sigma_{pc} A_{pc}} {R_b b} \end{equation} 
:x
(fnk
 ""
 [Rb b As Asc-ef Ap Apc Rs Rp Rsc sigma-pc]
 (/ (+ (* Rp Ap)
       (* Rs As)
       (* -1 Rsc Asc-ef)
       (* -1 sigma-pc Apc))
    (* Rb b)))

1.2.3 \(h_0\), \(h_{01}\) — —Ä–∞–±–æ—á–∞—è –≤—ã—Å–æ—Ç–∞ —Å–µ—á–µ–Ω–∏—è

–†–∞–±–æ—á–∞—è –≤—ã—Å–æ—Ç–∞ –¥–ª—è –Ω–µ–Ω–∞–ø—Ä—è–≥–∞–µ–º–æ–π –∞—Ä–º–∞—Ç—É—Ä—ã —Ä–∞–≤–Ω–∞ \(h_{01} = h - a_s\).

–†–∞–±–æ—á–∞—è –≤—ã—Å–æ—Ç–∞ —Å–µ—á–µ–Ω–∏—è \(h_0\) –ø—Ä–∏–Ω–∏–º–∞–µ—Ç—Å—è –æ—Ç —Ä–∞–≤–Ω–æ–¥–µ–π—Å—Ç–≤—É—é—â–∏–π —É—Å–∏–ª–∏–π –≤ –Ω–∞–ø—Ä—è–≥–∞–µ–º–æ–π –∏ –Ω–µ–Ω–∞–ø—Ä—è–≥–∞–µ–º–æ–π –∞—Ä–º–∞—Ç—É—Ä–µ.

\begin{equation} h_0 = \cfrac{A_s R_s h_{01} + A_p R_p (h - a_p)}{A_s R_s + A_p R_p} \end{equation}

–ü—Ä–∏ –æ—Ç—Å—É—Ç—Å—Ç–≤–∏–∏ –Ω–∞–ø—Ä—è–≥–∞–µ–º–æ–π –∞—Ä–º–∞—Ç—É—Ä—ã \(h_0 = h_{01}\). 

:h01
(fnk
 [h as]
 (- h as))

:h0
(fnk
 [h h01 ap As Rs Ap Rp]
 (if (zero? Ap)
   h01
   (/ (+ (* As Rs h01)
         (* Ap Rp (- h ap)))
      (+ (* As Rs)
         (* Ap Rp)))))

1.2.4 –£—á–µ—Ç –Ω–∞–ø—Ä—è–≥–∞–µ–º–æ–π —Å–∂–∞—Ç–æ–π –∞—Ä–º–∞—Ç—É—Ä—ã

–ù–∞–ø—Ä—è–≥–∞–µ–º—É—é –∞—Ä–º–∞—Ç—É—Ä—É, —Ä–∞—Å–ø–ª–æ–∂–æ–∂–µ–Ω–Ω—É—é –≤ —Å–∂–∞—Ç–æ–π –∑–æ–Ω–µ –∏ –∏–º–µ—é—â—É—é —Å—Ü–µ–ø–ª–µ–Ω–∏–µ —Å –±–µ—Ç–æ–Ω–æ–º, —Å–ª–µ–¥—É–µ—Ç –≤–≤–æ–¥–∏—Ç—å –≤ —Ä–∞—Å—á—ë—Ç —Å –Ω–∞–ø—Ä—è–∂–µ–Ω–∏–µ–º:

\begin{equation} \sigma_{pc} = R_{pc} - \sigma_{pc1} \end{equation}

–≥–¥–µ:

  • \(R_{pc}\) — —É—á–∏—Ç—ã–≤–∞–µ–º–æ–µ —Ä–∞—Å—á—ë—Ç–æ–º –Ω–∞–∏–±–æ–ª—å—à–µ–µ —Å–∂–∏–º–∞—é—â–µ–µ –Ω–∞–ø—Ä—è–∂–µ–Ω–∏–µ –≤ –Ω–∞–ø—Ä—è–≥–∞–µ–º–æ–π –∞—Ä–º–∞—Ç—É—Ä–µ;
  • \(\sigma_{pc1}\) — —Ä–∞—Å—á—ë—Ç–Ω–æ–µ –Ω–∞–ø—Ä—è–∂–µ–Ω–∏–µ –≤ –Ω–∞–ø—Ä—è–≥–∞–µ–º–æ–π –∞—Ä–º–∞—Ç—É—Ä–µ (–∑–∞ –≤—ã—á–µ—Ç–æ–º –≤—Å–µ—Ö –ø–æ—Ç–µ—Ä—å) –ø—Ä–∏ –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç–µ –Ω–∞–¥–µ–∂–Ω–æ—Å—Ç–∏ \(\gamma_g = 1.1\); –ø—Ä–∏ \(\sigma_{pc} \le R_{pc}\) –ø—Ä–∏–Ω–∏–º–∞–µ—Ç—Å—è \(\sigma_{pc} = 0\). 
:sigma-pc
(fnk
 ""
 [Rpc sigma-pc1]
 (if (<= sigma-pc1 Rpc)
   (si/MPa 0.0)
   (- Rpc sigma-pc1)))

(let [lazy-rect (lazy-compile rect-bending-flow)
      lazy-xi   (lazy-compile xi-flow)]
  (def rect-bending
    (fnk
     {:keys [Rb b h Rs As as] :as args}
     (let [input (merge {:Rsc Rs :Asc ((pow si/m 2) 0) :asc (si/m 0)
                         :Ap ((pow si/m 2) 0) :Apc ((pow si/m 2) 0)
                         :Rp  (si/MPa 0) :Rpc (si/MPa 500)
                         :ap (si/m 0) :apc (si/m 0)
                         :sigma-pc1 (si/MPa 0)
                         :sigma-p (si/MPa 0)}
                        args)
           no-Asc  (lazy-rect (assoc input :Asc-ef ((pow si/m 2) 0)))
           all-Asc (lazy-rect (assoc input :Asc-ef (:Asc input)))]
       (lazy-xi (cond
                 (<  (:x no-Asc) (* 2 (:asc input))) (dissoc  no-Asc :M-max-sc)
                 (>= (:x all-Asc)(* 2 (:asc input))) (dissoc all-Asc :M-max-sc)
                 :else (assoc (dissoc no-Asc :M-max)
                              :x (* 2 (:asc input)))))))))
